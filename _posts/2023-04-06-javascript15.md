__고유 이벤트 중단__  

***  

event.preventDefault() 사용
```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <style>
        ul {padding: 0;}
        ul > li {list-style: none; display: inline-block; border:1px solid #777; padding: 0 5px;}
        a {text-decoration: none; color:black;}
    </style>
</head>
<body>
    <div>
        <ul class="pageNation">
            <li><a href="getList.board">1</a></li>
            <li><a href="getList.board">2</a></li>
            <li><a href="getList.board">3</a></li>
        </ul>
    </div>

    <script>
        var pageNation = document.querySelector(".pageNation")
        pageNation.onclick = function(){

            event.preventDefault();                 //a 태그의 고유이벤트가 중단됨.

            if(event.target.tagName != "A") return;

            var pageNum = event.target.innerHTML;
            console.log(pageNum);
        }
    </script>
</body>
</html>
```
